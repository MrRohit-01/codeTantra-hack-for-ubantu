def insertion_sort(n, arr):
    for i in range(1, n):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    print(*arr, end=".")

# Input
n = int(input().strip())
arr = list(map(int, input().strip().split()))

insertion_sort(n, arr)
